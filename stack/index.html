Jenkinsfile

pipeline {
  environment {
    registry = "buildarium/bdapi-kits"
    registryCredential = 'dockerhub'
    dockerImage = ''
  }

  agent any

  stages {
    stage('Test') {
      steps {
        checkout scm
        sh dotnet test
      }
    }
    stage('Build') {
      steps {
        script {
          dockerImage = docker.build registry
        }
      }
    }
    stage('Deploy') {
      steps{
        script {
          docker.withRegistry('', registryCredential) {
            dockerImage.push("${env.BUILD_NUMBER}")
            if (env.BRANCH_NAME == 'master') {
              dockerImage.push("latest")
              sh "kubectl set image deployment/bdapi-auth \
                  bdapi-kits=buildarium/bdapi-auth:${env.BUILD_NUMBER} \
                  --kubeconfig /var/lib/jenkins/config"
            } else if (env.BRANCH_NAME == 'develop') {
              dockerImage.push("dev")
            }
          }
        }
        sh "docker rmi $registry:$BUILD_NUMBER"
      }
    }
  }
}

  • registry = Docker image name
  • registryCredential = Docker Hub credentials saved to Jenkins
  • dockerImage = global variable to be set and get in multiple steps

Ambassador routes traffic to proper microservices based off of prefixes:

Kube YAML

---
apiVersion: v1
kind: Service
metadata:
  name: bdapi-kits
  annotations:
    getambassador.io/config: |
      ---
      apiVersion: ambassador/v1
      kind:  Mapping
      name:  bdapi-kits_mapping
      prefix: /kit/
      rewrite: /kit/
      service: bdapi-kits
spec:
  selector:
    app: bdapi-kits
  ports:
  - port: 80
    name: http-bdapi-kits
    targetPort: http-api
---
apiVersion: extensions/v1beta1
kind: Deployment
metadata:
  name: bdapi-kits
spec:
  replicas: 1
  strategy:
    type: RollingUpdate
  template:
    metadata:
      labels:
        app: bdapi-kits
    spec:
      containers:
      - name: bdapi-kits
        image: buildarium/bdapi-kits:latest
        imagePullPolicy: "Always"
        env:
        - name: ENV
          value: "prod"
        - name: DBUSER
          valueFrom:
            secretKeyRef:
              name: bdapi-secret
              key: neouser
        - name: DBPASS
          valueFrom:
            secretKeyRef:
              name: bdapi-secret
              key: neopass
        ports:
        - name: http-api
          containerPort: 80
